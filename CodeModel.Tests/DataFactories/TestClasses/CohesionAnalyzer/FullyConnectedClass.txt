namespace CodeModel.Tests.DataFactories.TestClasses.ProjectAnalyzer
{
    public class DisconnectedClass
    {
        private int a;
        private int b;
        private int c;

        public int APlusBPlusC()
        {
            return a + b + c;
        }

        public int BPlusCMinusA()
        {
            return b + c - a;
        }

        public int BMinusCPlusA()
        {
            return b - c + a;
        }
    }
}
